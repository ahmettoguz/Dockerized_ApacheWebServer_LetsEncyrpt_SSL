# BUILD IMAGE
  docker-compose build

# UP CONTAINERS
  docker-compose up

  detach mode:
  docker-compose up -d

# BUILD IMAGE AND UP CONTAINER (kod değişikliğinde böyle kaldır)
  docker-compose up -d --build

# DOWN & DELETE CONTAINERS (volumes are safe)
  docker-compose down

# STOP CONTAINERS
  docker-compose stop

# DISPLAY CONTAINERS
  docker ps -a

# GET IN CONTAINER
  docker exec -it c-util /bin/bash

# COPY FROM LOCAL 
  docker cp ./nginx/default.conf nginx-container:/

# COPY FROM CONTAINER
  docker cp util-c: /a.txt ./

# GET INTO CONTAINER
  docker exec -it php-c /bin/bash
  docker exec -it util-c /bin/bash

# SHOW 
  docker volume ls

  docker volume inspect storageVolume
  // mountpoint display the actual location of the volume

  docker volume create storageVolume

# REMOVE
  docker volume rm storageVolume
  docker volume prune
